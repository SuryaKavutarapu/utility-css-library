HTML Usage

<!-- Automatic contrast calculation -->
<button class="btn btn-warning">
  Warning (Black Text)
</button>

<button class="btn btn-doom">
  Doom (White Text)
</button>

<!-- Material Design Icons -->
<span class="mdi-icon mdi-icon--lg" 
      data-icon="heart"></span>

<!-- Opacity variants -->
<div class="bg-primary-30 p-4">
  30% opacity background
</div>

TypeScript Usage

import { ThemeManager } from 'utility-css-library';

// Initialize theme management
const themeManager = new ThemeManager();

// Toggle theme
themeManager.toggleMode();

// Get current theme
const mode = themeManager.getCurrentMode();

// Subscribe to changes
themeManager.subscribe((newMode) => {
  console.log(`Theme: ${newMode}`);
});
